SELECT pa.nom, pa.prenom, concat(pa.rue,' ',pa.code_postal,' ',pa.commune) as address, pa.telephone, pa.gsm, pa.mail, mp.description as description, pi.date_derniere_modification, pi.statut, pi.id_motif,pi.id_patient FROM mp_pile pi, patients pa, medecine_preventive mp WHERE statut='termine' AND pa.id = pi.id_patient AND mp.motif_ID = pi.id_motif  ORDER BY statut asc
